    function EMACalculator($limit,$array)
    {
        $EMA_previous_day = $array[0];
        //print_r($array);
        $multiplier1 = (2/$limit+1);
        $EMA[]=array();
        $EMA = $array[0];
        $Close= $array[1];
       
        while($limit)
    {   
        //echo"EMA is $EMA\n";
        $EMA = ($Close - $EMA_previous_day) * $multiplier1 + $EMA_previous_day;
        $EMA_previous_day= $EMA;
       
        $limit--;
    }
    return $EMA;
    }
