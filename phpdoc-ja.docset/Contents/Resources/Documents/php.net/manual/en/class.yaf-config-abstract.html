abstract Yaf_Config_Abstract {

protected $_readonly ;
/* メソッド */
abstract public mixed get ( string $name , mixed $value )
abstract public bool readonly ( void )
abstract public Yaf_Config_Abstract set ( void )
abstract public array toArray ( void )
}
